54 static inline __attribute__((always_inline)) void IOPinSetDir(int PortNo, int PinNo, IOPINDIR Dir) {
55  if (Dir == IOPINDIR_OUTPUT)
56  LPC_GPIO->DIR[PortNo] |= 1 << PinNo;
57  else if (Dir == IOPINDIR_INPUT)
58  LPC_GPIO->DIR[PortNo] &= ~(1 << PinNo);
59 }
