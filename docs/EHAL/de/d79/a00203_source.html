163 void ADVEventHandler(ADCDevice *pAdcDev, ADC_EVT Evt)
164 {
165  if (Evt == ADC_EVT_DATA_READY)
166  {
167  g_bDataReady = true;
168 #ifdef ADC_DEMO_INTERRUPT_ENABLE
169  int cnt = 0;
170 
171  ADC_DATA df[s_NbChan];
172  cnt = g_Adc.Read(df, s_NbChan);
173  if (cnt > 0)
174  {
175  g_Uart.printf("%d ADC[0] = %.2fV, ADC[1] = %.2fV, ADC[2] = %.2fV, ADC[3] = %.2fV\r\n",
176  df[0].Timestamp, df[0].Data, df[1].Data, df[2].Data, df[3].Data);
177  }
178 
179  if (g_Adc.Mode() == ADC_CONV_MODE_SINGLE)
180  {
181  g_Adc.StartConversion();
182  }
183 #endif
184  }
185 }
