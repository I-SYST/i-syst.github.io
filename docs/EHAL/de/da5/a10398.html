Environmental sensor demo.

This application demo shows how to use environmental driver (BME280, BME680, MS8607). Temperature, Pressure, Humidity (TPH). Support I2C or SPI interface. Data is printed to UART.

extern "C" void bme680_bsec_read_data(int64_t time_stamp_trigger, bsec_input_t *input, uint8_t *num_bsec_inputs,
int32_t bsec_process_data)
{
TPHSENSOR_DATA tphdata;
GASSENSOR_DATA gdata;
g_EnvSensor.StartSampling();
usDelay(200000);
g_EnvSensor.Read(tphdata);
bool res = g_EnvSensor.Read(gdata);
//if (res == false)
// return;
uint8_t nbi = 0;
time_stamp_trigger = tphdata.Timestamp * 1000LL;
if (bsec_process_data & BSEC_PROCESS_TEMPERATURE)
{
input[nbi].sensor_id = BSEC_INPUT_TEMPERATURE;
input[nbi].signal = tphdata.Temperature / 100.0;
input[nbi].time_stamp= time_stamp_trigger;
nbi++;
}
if (bsec_process_data & BSEC_PROCESS_HUMIDITY)
{
input[nbi].sensor_id = BSEC_INPUT_HUMIDITY;
input[nbi].signal = tphdata.Humidity / 100.0;
input[nbi].time_stamp= time_stamp_trigger;
nbi++;
}
if (bsec_process_data & BSEC_PROCESS_PRESSURE)
{
input[nbi].sensor_id = BSEC_INPUT_PRESSURE;
input[nbi].signal = tphdata.Pressure;
input[nbi].time_stamp= time_stamp_trigger;
nbi++;
}
if (bsec_process_data & BSEC_PROCESS_GAS)
{
if (res == true)
{
input[nbi].sensor_id = BSEC_INPUT_GASRESISTOR;
input[nbi].signal = gdata.GasRes[gdata.MeasIdx];
input[nbi].time_stamp = time_stamp_trigger;// g_Timer.mSecond();
nbi++;
}
}
*num_bsec_inputs = nbi;
//g_EnvSensor.StartSampling();
}
