29  // DFU Write callback function.
30  //
31  // This function is provided by the application software. This function gets
32  // called when host sends a write command. For application using zero-copy
33  // buffer scheme this function is called for the first time with length
34  // parameter set to 0. The application code should update the buffer pointer.
35  //
36  // Parameters:
37  // [in] block_num Destination start address.
38  // [in,out] src Pointer to a pointer to the source of data.
39  // Pointer-to-pointer is used to implement zero-copy
40  // buffers. See Zero-Copy Data Transfer model for more
41  // details on zero-copy concept.
42  // [out] bwPollTimeout Pointer to a 3 byte buffer which the callback implementer should fill with the amount of minimum time, in milliseconds, that the host should wait before sending a subsequent DFU_GETSTATUS request.
43  // [in] length Number of bytes to be written.
44  // Returns:
45  // Returns DFU_STATUS_ values defined in mw_usbd_dfu.h.
46  uint8_t (*DFU_Write )(uint32_t block_num, uint8_t **src, uint32_t length, uint8_t *bwPollTimeout);
47 
48  // DFU Read callback function.
49  //
50  // This function is provided by the application software. This function gets
51  // called when host sends a read command.
52  //
53  // Parameters:
54  // [in] block_num Destination start address.
55  // [in,out] dst Pointer to a pointer to the source of data.
56  // Pointer-to-pointer is used to implement zero-copy buffers.
57  // See Zero-Copy Data Transfer model for more details on
58  // zero-copy concept.
59  // [in] length Amount of data copied to destination buffer.
60  // Returns:
61  // DFU_STATUS_ values defined in mw_usbd_dfu.h to return error conditions.
62  // 0 if there is no more data to be read. Stack will send EOF frame and
63  // set DFU state-machine to dfuIdle state.
64  // length of the data copied, should be greater than or equal to 16. If
65  // the data copied is less than DFU wTransferSize the stack will send EOF
66  // frame and goes to dfuIdle state.
67  uint32_t (*DFU_Read )(uint32_t block_num, uint8_t **dst, uint32_t length);
68 
69  // DFU done callback function.
70  //
71  // This function is provided by the application software. This function gets
72  // called after firmware download completes.
73  //
74  // Returns:
75  // Nothing.
76  void (*DFU_Done )(void);
77 
78  // DFU detach callback function.
79  //
80  // This function is provided by the application software. This function gets
81  // called after USB_REQ_DFU_DETACH is received. Applications which set
82  // USB_DFU_WILL_DETACH bit in DFU descriptor should define this function. As
83  // part of this function application can call Connect() routine to disconnect
84  // and then connect back with host. For application which rely on WinUSB based
85  // host application should use this feature since USB reset can be invoked
86  // only by kernel drivers on Windows host. By implementing this feature host
87  // doen't have to issue reset instead the device has to do it automatically by
88  // disconnect and connect procedure.
89  //
90  // Parameters:
91  // [in] hUsb Handle DFU control structure.
92  // Returns:
93  // Nothing.
94  void (*DFU_Detach)(USBD_HANDLE_T hUsb);
95 
96  // Optional user override-able function to replace the default DFU class handler.
97  //
98  // The application software could override the default EP0 class handler with
99  // their own by providing the handler function address as this data member of
100  // the parameter structure. Application which like the default handler should
101  // set this data member to zero before calling the USBD_DFU_API::Init().
102  // Note:
103  // Parameters:
104  // [in] hUsb Handle to the USB device stack.
105  // [in] data Pointer to the data which will be passed when callback
106  // function is called by the stack.
107  // [in] event Type of endpoint event. See USBD_EVENT_T for more details.
108  // Returns:
109  // The call back should returns ErrorCode_t type to indicate success or
110  // error condition.
111  // LPC_OK On success.
112  // ERR_USBD_UNHANDLED Event is not handled hence pass the event to next in line.
113  // ERR_USBD_xxx For other error conditions.
114  uint32_t (*DFU_Ep0_Hdlr)(USBD_HANDLE_T hUsb, void *data, uint32_t event);
115 } USBD_DFU_INIT_PARAM_T;

121  // This function is called by application layer before calling pUsbApi->dfu->Init(),

135  // This function is called by application layer to initialize DFU function

137  //
138  // Parameters:
139  // [in] hUsb Handle to the USB device stack.
140  // [in,out] param Structure containing DFU function driver module
141  // initialization parameters.
142  // Returns:
143  // LPC_OK On success
144  // ERR_USBD_BAD_MEM_BUF Memory buffer passed is not 4-byte aligned or
