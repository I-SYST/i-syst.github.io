56  // Event callback for USB interface reset. This event fires when the USB host requests that the
57  // device reset its interface. This event fires after the control endpoint has been
58  // automatically configured by the library.
59  // Note:
60  // This event is called from USB_ISR context and hence is time-critical. Having delays
61  // in this callback will prevent the device from enumerating correctly or operate properly.
62  uint32_t (*USB_Reset_Event)(USBD_HANDLE_T hUsb);
63 
64  // Event for USB suspend. This event fires when the USB host suspends the device by
65  // halting its transmission of Start Of Frame pulses to the device. This is generally
66  // hooked in order to move the device over to a low power state until the host wakes
67  // up the device.
68  // Note:
69  // This event is called from USB_ISR context and hence is time-critical. Having delays
70  // in this callback will cause other system issues.
71  uint32_t (*USB_Suspend_Event)(USBD_HANDLE_T hUsb);
72 
73  // Event for USB wake up or resume. This event fires when a the USB device interface
74  // is suspended and the host wakes up the device by supplying Start Of Frame pulses.
75  // This is generally hooked to pull the user application out of a low power state and
76  // back into normal operating mode.
77  // Note:
78  // This event is called from USB_ISR context and hence is time-critical. Having delays
79  // in this callback will cause other system issues.
80  uint32_t (*USB_Resume_Event)(USBD_HANDLE_T hUsb);
81 
82  void* reserved_sbz; // Reseved, must be set to 0
83 
84  // Event for USB Start Of Frame detection, when enabled. This event fires at the start
85  // of each USB frame, once per millisecond in full-speed mode or once per 125
86  // microseconds in high-speed mode, and is synchronized to the USB bus.
87  //
88  // This event is time-critical; it is run once per millisecond (full-speed mode) and
89  // thus long handlers will significantly degrade device performance. This event should
90  // only be enabled when needed to reduce device wake-ups.
91  //
92  // Note:
93  // This event is not normally active - it must be manually enabled and disabled via the
94  // USB interrupt register.
95  uint32_t (*USB_SOF_Event)(USBD_HANDLE_T hUsb);
96 
97  // Event for remote wake-up configuration, when enabled. This event fires when the USB
98  // host request the device to configure itself for remote wake-up capability. The USB
99  // host sends this request to device which report remote wake-up capable in their
100  // device descriptors, before going to low-power state. The application layer should
101  // implement this callback if they have any special on board circuit to trigger remote
102  // wake up event. Also application can use this callback to differentiate the following
103  // SUSPEND event is caused by cable plug-out or host SUSPEND request. The device can
104  // wake-up host only after receiving this callback and remote wake-up feature is
105  // enabled by host. To signal remote wake-up the device has to generate resume signaling
106  // on bus by calling usapi.hw->WakeUp() routine.
107  //
108  // Parameters:
109  // [in] hUsb Handle to the USB device stack.
110  // [in] param1 When 0 - Clear the wake-up configuration,
111  // 1 - Enable the wake-up configuration.
112  // Returns:
113  // The call back should return ErrorCode_t type to indicate success or error condition.
114  uint32_t (*USB_WakeUpCfg)(USBD_HANDLE_T hUsb, uint32_t param1);
115 
116  uint32_t (*USB_Power_Event)(USBD_HANDLE_T hUsb, uint32_t param1); // Reserved parameter should be set to zero.
117 
118  // Event for error condition. This event fires when USB device controller detect an
119  // error condition in the system.
120  //
121  // Parameters:
122  // [in] hUsb Handle to the USB device stack.
123  // [in] param1 USB device interrupt status register.
124  // Returns:
125  // The call back should return ErrorCode_t type to indicate success or error condition.
126  uint32_t (*USB_Error_Event)(USBD_HANDLE_T hUsb, uint32_t param1);

128  // Event for USB configuration number changed. This event fires when a the USB host changes
129  // the selected configuration number. On receiving configuration change request from host,
130  // the stack enables/configures the endpoints needed by the new configuration before calling
131  // this callback function.
132  // Note:
133  // This event is called from USB_ISR context and hence is time-critical. Having delays in this
134  // callback will prevent the device from enumerating correctly or operate properly.
135  uint32_t (*USB_Configure_Event)(USBD_HANDLE_T hUsb);
136 
137  // Event for USB interface setting changed. This event fires when a the USB host changes the
138  // interface setting to one of alternate interface settings. On receiving interface change
139  // request from host, the stack enables/configures the endpoints needed by the new alternate
140  // interface setting before calling this callback function.
141  // Note:
142  // This event is called from USB_ISR context and hence is time-critical. Having delays in
143  // this callback will prevent the device from enumerating correctly or operate properly.
144  uint32_t (*USB_Interface_Event)(USBD_HANDLE_T hUsb);
145 
146  // Event for USB feature changed. This event fires when a the USB host send set/clear feature
147  // request. The stack handles this request for USB_FEATURE_REMOTE_WAKEUP, USB_FEATURE_TEST_
148  // MODE and USB_FEATURE_ENDPOINT_STALL features only. On receiving feature request from host,
149  // the stack handle the request appropriately and then calls this callback function.
150  // Note:
151  // This event is called from USB_ISR context and hence is time-critical. Having delays in this
152  // callback will prevent the device from enumerating correctly or operate properly.
153  uint32_t (*USB_Feature_Event)(USBD_HANDLE_T hUsb);
