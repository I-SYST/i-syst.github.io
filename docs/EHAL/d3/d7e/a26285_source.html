181 
182  // DevCs is the ordinal starting from 0 of device connected to the SPI bus.
183  // It is translated to CS index in the I/O pin map
184  virtual bool StartRx(int DevCs) {
185  return DeviceIntrfStartRx(&vDevData.DevIntrf, DevCs);
186  }
187  // Receive Data only, no Start/Stop condition
188  virtual int RxData(uint8_t *pBuff, int BuffLen) {
189  return DeviceIntrfRxData(&vDevData.DevIntrf, pBuff, BuffLen);
190  }
191  virtual void StopRx(void) { DeviceIntrfStopRx(&vDevData.DevIntrf); }
192  // DevAddr is the ordinal starting from 0 of device connected to the SPI bus.
193  // It is translated to CS index in the I/O pin map
194  virtual bool StartTx(int DevCs) {
195  return DeviceIntrfStartTx(&vDevData.DevIntrf, DevCs);
196  }
197  // Send Data only, no Start/Stop condition
198  virtual int TxData(uint8_t *pData, int DataLen) {
199  return DeviceIntrfTxData(&vDevData.DevIntrf, pData, DataLen);
200  }
201  virtual void StopTx(void) { DeviceIntrfStopTx(&vDevData.DevIntrf); }
202  int GetFirstRead(void) {return vDevData.FirstRdData;}
203 
204 private:
