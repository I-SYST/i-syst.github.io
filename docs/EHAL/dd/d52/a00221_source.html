320 void ReadPTHData()
321 {
322  static uint32_t gascnt = 0;
323  TPHSENSOR_DATA data;
324 
325  g_TphSensor.Read(data);
326 
327 
328 
329  if (g_TphSensor.DeviceID() == BME680_ID && (gascnt & 0x3) == 0)
330  {
331  GASSENSOR_DATA gdata;
332  BLEADV_MANDATA_GASSENSOR gas;
333 
334  g_GasSensor.Read(gdata);
335 
336  g_AdvData.Type = BLEADV_MANDATA_TYPE_GAS;
337  gas.GasRes = gdata.GasRes[gdata.MeasIdx];
338  gas.AirQIdx = gdata.AirQualIdx;
339 
340  memcpy(&g_GasData, &gas, sizeof(BLEADV_MANDATA_GASSENSOR));
341  }
342  else
343  {
344  g_AdvData.Type = BLEADV_MANDATA_TYPE_TPH;
345 
346  // NOTE : M0 does not access unaligned data
347  // use local 4 bytes align stack variable then mem copy
348  // skip timestamp as advertising pack is limited in size
349  memcpy(&g_TPHData, ((uint8_t*)&data) + 8, sizeof(BLEADV_MANDATA_TPHSENSOR));
350  }
351 
352  g_TphSensor.StartSampling();
353 
354  // Update advertisement data
355  BleAppAdvManDataSet(g_AdvDataBuff, 8);
356 
357  BleSrvcCharNotify(&g_TphBleSrvc, 0, g_AdvDataBuff, sizeof(g_AdvDataBuff));
358  //BleSrvcCharSetValue(&g_TphBleSrvc, 0, g_AdvDataBuff, sizeof(g_AdvDataBuff));
359 
360  gascnt++;
361 }

398 void HardwareInit()

400  // Set this only if nRF is power at 2V or more

402 
403  g_Timer.Init(s_TimerCfg);
404

409  g_I2c.Init(s_I2cCfg);
410 #endif
411 
412 
413  // Inititalize sensor
414  g_TphSensor.Init(s_TphSensorCfg, g_pIntrf, &g_Timer);
415 
416  if (g_TphSensor.DeviceID() == BME680_ID)
417  {
418  g_GasSensor.Init(s_GasSensorCfg, g_pIntrf, NULL);
419  }
420 
421  g_TphSensor.StartSampling();
422 
423  usDelay(200000);
424 
425  // Update sensor data
426  TPHSENSOR_DATA tphdata;
427 
428  g_TphSensor.Read(tphdata);
429 
430  if (g_TphSensor.DeviceID() == BME680_ID)
431  {
432  GASSENSOR_DATA gdata;
433  g_GasSensor.Read(gdata);
434  }
435 
436  g_TphSensor.StartSampling();
437 
438  g_AdvData.Type = BLEADV_MANDATA_TYPE_TPH;
439  // Do memcpy to adv data. Due to byte alignment, cannot read directly into
440  // adv data
441  memcpy(g_AdvData.Data, ((uint8_t*)&tphdata) + 4, sizeof(BLEADV_MANDATA_TPHSENSOR));
