56 // IIR
57 #define LPCUART_IIR_STATUS 1 // IntStatus Interrupt status. Note that UnIIR[0] is active low.
58  // 0 At least one interrupt is pending.
59  // 1 No interrupt is pending.
60 #define LPCUART_IIR_ID_MASK (7<<1) // IntId Interrupt identification.
61  // 011 1 - Receive Line Status (RLS).
62  // 010 2a - Receive Data Available (RDA).
63  // 110 2b - Character Time-out Indicator (CTI).
64  // 001 3 - THRE Interrupt
65  // 000 4 - Modem status
66 #define LPCUART_IIR_ID_MS (0<<1) // Modem status
67 #define LPCUART_IIR_ID_THRE (1<<1) // THRE Interrupt.
68 #define LPCUART_IIR_ID_RDA (2<<1) // Receive Data Available (RDA)
69 #define LPCUART_IIR_ID_RLS (3<<1) // Receive Line Status (RLS).
70 #define LPCUART_IIR_ID_CTIMOUT (6<<1) // Character Time-out Indicator (CTI).
71 
72 #define LPCUART_IIR_FIFO_MASK 0xc0 // FIFO Enable Copies of UnFCR[0].
73 #define LPCUART_IIR_ABEO 0x100 // End of auto-baud interrupt. True if auto-baud has finished successfully and
74  // interrupt is enabled.
75 
76 #define LPCUART_IIR_ABTO 0x200 // Auto-baud time-out interrupt. True if auto-baud has timed out and interrupt is
77  // enabled.

79 // FCR - FIFO Control Register
80 #define LPCUART_FCR_FIFOEN 1 // 0 - UARTn FIFOs are disabled. Must not be used in the application.
81  // 1 Active high enable for both UARTn Rx and TX FIFOs and UnFCR[7:1] access.
82  // This bit must be set for proper UART operation. Any transition on this bit will
83  // automatically clear the related UART FIFOs.
84 #define LPCUART_FCR_RST_RXFIFO 2 // 0 - No impact on either of UARTn FIFOs.
85  // 1 - Writing a logic 1 to UnFCR[1] will clear all bytes in UARTn Rx FIFO, reset the
86  // pointer logic. This bit is self-clearing.
87 #define LPCUART_FCR_RST_TXFIFO 4 // 0 - No impact on either of UARTn FIFOs.
88  // 1 - Writing a logic 1 to UnFCR[2] will clear all bytes in UARTn TX FIFO, reset the
89  // pointer logic. This bit is self-clearing.
90 #define LPCUART_FCR_DMA_MODE 8 // DMA Mode
91 #define LPCUART_FCR_RX_TRIG_MASK (3<<6) // RX Trigger Level. These two bits determine how many receiver UARTn FIFO characters must be
92  // written before an interrupt or DMA request is activated.
93 #define LPCUART_FCR_RX_TRIG1 (0<<6) // 00 Trigger level 0 (1 character or 0x01)
94 #define LPCUART_FCR_RX_TRIG4 (1<<6) // 01 Trigger level 1 (4 characters or 0x04)
95 #define LPCUART_FCR_RX_TRIG8 (2<<6) // 10 Trigger level 2 (8 characters or 0x08)
96 #define LPCUART_FCR_RX_TRIG14 (3<<6) // 11 Trigger level 3 (14 characters or 0x0E)
97 
98 // LCR - Line Control Register
99 #define LPCUART_LCR_WLEN_MASK 3 // Word Length Select
100  // 00 5-bit character length
101  // 01 6-bit character length
102  // 10 7-bit character length
103  // 11 8-bit character length
104 #define LPCUART_LCR_STOPBIT_MASK 4 // Stop Bit Select
105  // 0 : 1 stop bit. 0
106  // 1 : 2 stop bits (1.5 if UnLCR[1:0]=00).
107 #define LPCUART_LCR_PAR_EN 8 // Parity Enable 0 Disable parity generation and checking.
108  // 1 Enable parity generation and checking.
109 #define LPCUART_LCR_PAR_MASK 0x30 // Parity Select
110  // 00 Odd parity. Number of 1s in the transmitted character and the attached
111  // parity bit will be odd.
112  // 01 Even Parity. Number of 1s in the transmitted character and the attached
113  // parity bit will be even.
114  // 10 Forced "1" stick parity.
115  // 11 Forced "0" stick parity.
116 #define LPCUART_LCR_BRK 0x40 // Disable break transmission.
117  // 1 Enable break transmission. Output pin UARTn TXD is forced to logic 0
118  // when UnLCR[6] is active high.
119 #define LPCUART_LCR_DLAB 0x80 // Divisor Latch Access Bit (DLAB)
120  // 0 - Disable access to Divisor Latches.
121  // 1 - Enable access to Divisor Latches.
122 
123 // MCR - Modem control register
124 #define LPCUART_MCR_DTRCTRL (1<<0) // Source for modem output pin DTR. This bit reads as 0 when modem loopback
125  // mode is active
126 #define LPCUART_MCR_RTSCTRL (1<<1) // Source for modem output pin RTS. This bit reads as 0 when modem loopback
127  // mode is active.
128 #define LPCUART_MCR_LMS (1<<4) // Loopback Mode Select.
129 #define LPCUART_MCR_RTSEN (1<<6) // RTS enable
130 #define LPCUART_MCR_CTSEN (1<<7) // CTS enable
131 
132 // LSR - Line Status Register
133 #define LPCUART_LSR_RDR 1 // Receiver Data Ready (RDR)
134  // UnLSR0 is set when the UnRBR holds an unread character and is cleared when
135  // the UARTn RBR FIFO is empty.
136  // 0 The UARTn receiver FIFO is empty.
137  // 1 The UARTn receiver FIFO is not empty.
138 #define LPCUART_LSR_OE 2 // Overrun Error (OE)
139  // The overrun error condition is set as soon as it occurs. An UnLSR read clears
140  // UnLSR1. UnLSR1 is set when UARTn RSR has a new character assembled and
141  // the UARTn RBR FIFO is full. In this case, the UARTn RBR FIFO will not be
142  // overwritten and the character in the UARTn RSR will be lost.
143  // 0 Overrun error status is inactive.
144  // 1 Overrun error status is active.
145 #define LPCUART_LSR_PE 4 // Parity Error (PE)
146  // Note: A parity error is associated with the character at the top of the UARTn RBR FIFO.
147  // 0 Parity error status is inactive.
148  // 1 Parity error status is active.
149 #define LPCUART_LSR_FE 8 // Framing Error (FE)
150  // 0 Framing error status is inactive.
151  // 1 Framing error status is active.
152 #define LPCUART_LSR_BI 0x10 // Break Interrupt (BI)
153  // 0 Break interrupt status is inactive.
154  // 1 Break interrupt status is active.
155 #define LPCUART_LSR_THRE 0x20 // Transmitter Holding Register Empty (THRE))
156  // 0 UnTHR contains valid data.
157  // 1 UnTHR is empty.
158 #define LPCUART_LSR_TEMT 0x40 // Transmitter Empty (TEMT)
159  // 0 UnTHR and/or the UnTSR contains valid data.
160  // 1 UnTHR and the UnTSR are empty.
161 #define LPCUART_LSR_RXFE 0x80 // Error in RX FIFO (RXFE)
162  // 0 UnRBR contains no UARTn RX errors or UnFCR[0]=0.
163  // 1 UARTn RBR contains at least one UARTn RX error.
164 
165 #define LPCUART_MSR_DCTS 1 // Delta CTS. 0 Set upon state change of input CTS. Cleared on an MSR read.
166  // 0 - No change detected on modem input, CTS.
167  // 1 - State change detected on modem input, CTS.
168 #define LPCUART_MSR_DDSR 2 // Delta DSR. 0 Set upon state change of input DSR. Cleared on an MSR read.
169  // 0 - No change detected on modem input, DSR.
170  // 1 - State change detected on modem input, DSR.
171 #define LPCUART_MSR_TERI 4 // Trailing Edge RI. 0 Set upon low to high transition of input RI. Cleared on an
172  // MSR read.
173  // 0 - No change detected on modem input, RI.
174  // 1 - Low-to-high transition detected on RI.
175 #define LPCUART_MSR_DDCD 8 // Delta DCD. Set upon state change of input DCD. Cleared on 0 an MSR read.
176  // 0 - No change detected on modem input, DCD.
177  // 1 - State change detected on modem input, DCD.
178 #define LPCUART_MSR_CTS 0x10 // Clear To Send State. Complement of input signal CTS. This 0 bit is connected to MCR[1]
179  // in modem loopback mode.
180 #define LPCUART_MSR_DSR 0x20 // Data Set Ready State. Complement of input signal DSR. 0 This bit is connected to MCR[0]
181  // in modem loopback mode.
182 #define LPCUART_MSR_RI 0x40 // Ring Indicator State. Complement of input RI. This bit is 0 connected to MCR[2]
183  // in modem loopback mode.
184 #define LPCUART_MSR_DCD 0x80 // Data Carrier Detect State. Complement of input DCD. This 0 bit is connected to MCR[3]
185  // in modem loopback mode.
186 
187 // ACR - Auto-baud Control Register
188 #define LPCUART_ACR_START 1 // Start This bit is automatically cleared after auto-baud completion.
189  // 0 Auto-baud stop (auto-baud is not running).
190  // 1 Auto-baud start (auto-baud is running). Auto-baud run bit. This bit is
191  // automatically cleared after auto-baud completion.
192 #define LPCUART_ACR_MODE 2 // Mode Auto-baud mode select bit.
193  // 0 Mode 0.
194  // 1 Mode 1.
195 #define LPCUART_ACR_AUTORESTART 4 // AutoRestart 0 No restart.
196  // 1 Restart in case of time-out (counter restarts at next UARTn Rx falling edge) 0
197 #define LPCUART_ACR_ABEO_INTCLR 0x100 // ABEOIntClr End of auto-baud interrupt clear bit (write-only accessible). Writing a 1 will
198  // clear the corresponding interrupt in the UnIIR. Writing a 0 has no impact.
199 #define LPCUART_ACR_ABTO_INTCLR 0x200 // ABTOIntClr Auto-baud time-out interrupt clear bit (write-only accessible). Writing a 1 will
200  // clear the corresponding interrupt in the UnIIR. Writing a 0 has no impact.
201
