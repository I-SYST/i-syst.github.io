50 // I2C Status code
51 typedef enum _I2C_Status {
52  I2CSTATUS_START_COND = 8, // Start condition transmitted
53  I2CSTATUS_RESTART_COND = 0x10, // Start condition re-transmitted
54  I2CSTATUS_SLAW_ACK = 0x18, // SLA+W has been transmitted; ACK has been received
55  I2CSTATUS_SLAW_NACK = 0x20, // SLA+W has been transmitted; NO ACK has been received
56  I2CSTATUS_M_TXDAT_ACK = 0x28, // Data byte in I2DAT has been transmitted; ACK has been received
57  I2CSTATUS_M_TXDAT_NACK = 0x30, // Data byte in I2DAT has been transmitted; NO ACK has been received
58  I2CSTATUS_ARB_LOST = 0x38, // Arbitration lost in SLA+R/W or Data bytes
59  I2CSTATUS_SLAR_ACK = 0x40, // SLA+R has been transmitted; ACK has been received
60  I2CSTATUS_SLAR_NACK = 0x48, // SLA+R has been transmitted; NO ACK has been received
61  I2CSTATUS_RXDATA_ACK = 0x50, // Data byte has been received; ACK has been returned
62  I2CSTATUS_RXDATA_NACK = 0x58, // ata byte has been received; NO ACK has been returned
63  I2CSTATUS_OWNSLAW_ACK = 0x60, // Own SLA+W has been received; ACK has been returned
64  I2CSTATUS_ARB_LOST_OWNSLAW_NACK = 0x68, // Arbitration lost in SLA+R/W as master; Own SLA+W has been received, ACK returned
65  I2CSTATUS_GENCALL_ACK = 0x70, // General Call address (0x00) has been received; ACK has been returned
66  I2CSTATUS_ARB_LOST_GENCALL_NACK = 0x78, // Arbitration lost in SLA+R/W as master; General Call address has been received, ACK has been returned
67  I2CSTATUS_SLA_OWN_DATA_ACK = 0x80, // Previously addressed with own SLA address; DATA has been received; ACK has been returned
68  I2CSTATUS_SLA_OWN_DATA_NACK = 0x88, // Previously addressed with own SLA address; DATA has been received; NO ACK has been returned
69  I2CSTATUS_GENCALL_DATA_ACK = 0x90, // Previously addressed with General Call; DATA byte has been received; ACK has been returned
70  I2CSTATUS_GENCALL_DATA_NACK = 0x98, // Previously addressed with General Call; DATA byte has been received; NO ACK has been returned
71  I2CSTATUS_STOP_COND = 0xA0, // A STOP condition or repeated START condition has been received while still addressed as Slave Receiver or Slave Transmitter
72  I2CSTATUS_OWN_SLAR_ACK = 0xA8, // Own SLA+R has been received; ACK has been returned
73  I2CSTATUS_ARB_LOST_OWN_SLAR_ACK = 0xB0, // Arbitration lost in SLA+R/W as master; Own SLA+R has been received, ACK has been returned
74  I2CSTATUS_S_TXDAT_ACK = 0xB8, // Data byte in I2DAT has been transmitted; ACK has been received
75  I2CSTATUS_S_TXDAT_NACK = 0xC0, // Data byte in I2DAT has been transmitted; NOT ACK has been received
76  I2CSTATUS_S_LAST_TXDAT_ACK = 0xC8, // Last data byte in I2DAT has been transmitted (AA = 0); ACK has been received
77 } I2CSTATUS;
