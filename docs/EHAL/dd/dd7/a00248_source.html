78 class EsbIntrf : public DeviceIntrf {
79 public:
80  EsbIntrf();
81  virtual ~EsbIntrf();
82 
83  bool Init(const ESBINTRF_CFG &Cfg);
84 
85  operator DEVINTRF* () { return &vEsbIntrf.DevIntrf; } // No C implementation, just return NULL
86  // Set data rate in bits/sec (Hz)
87  int Rate(int DataRate) { return DeviceIntrfSetRate(&vEsbIntrf.DevIntrf, DataRate); }
88  // Get current data rate in bits/sec (Hz)
89  int Rate(void) { return DeviceIntrfGetRate(&vEsbIntrf.DevIntrf); }
90  // Disable device for power reduction, re-enable with Enable() without
91  // full init
92  void Disable(void) { DeviceIntrfDisable(&vEsbIntrf.DevIntrf); }
93  // Enable device
94  void Enable(void) { DeviceIntrfEnable(&vEsbIntrf.DevIntrf); }
95 
96  // Initiate receive
97  virtual bool StartRx(int DevAddr) { return DeviceIntrfStartRx(&vEsbIntrf.DevIntrf, DevAddr); }
98  // Receive Data only, no Start/Stop condition
99  virtual int RxData(uint8_t *pBuff, int BuffLen) {
100  return DeviceIntrfRxData(&vEsbIntrf.DevIntrf, pBuff, BuffLen);
101  }
102  // Stop receive
103  // BEWARE !!!!!
104  // This functions MUST ONLY be called if StartRx returns true.
105  virtual void StopRx(void) { DeviceIntrfStopRx(&vEsbIntrf.DevIntrf); }
106  // Initiate transmit
107  virtual bool StartTx(int DevAddr) {
108  return DeviceIntrfStartTx(&vEsbIntrf.DevIntrf, DevAddr);
109  }
110  // Transmit Data only, no Start/Stop condition
111  virtual int TxData(uint8_t *pData, int DataLen) {
112  return DeviceIntrfTxData(&vEsbIntrf.DevIntrf, pData, DataLen);
113  }
114  // Stop transmit
115  // BEWARE !!!!!
116  // This functions MUST ONLY be called if StartTx returns true.
117  virtual void StopTx(void) { DeviceIntrfStopTx(&vEsbIntrf.DevIntrf); }
118 
119 private:
120  ESBINTRF vEsbIntrf;
121 };
