96 typedef struct _USB_Msc_Command_Block_Wrapper {
97  uint32_t dCBWSignature; // The signature field shall contain the value
98  // 43425355h (little endian), indicating a CBW.
99  uint32_t dCBWTag; // A Command Block Tag sent by the host. The
100  // device shall echo the contents of this field
101  // back to the host in the dCSWTag field of the
102  // associated CSW
103  uint32_t dCBWDataTransferLength;// The number of bytes of data that the host
104  // expects to transfer on the Bulk-In or
105  // Bulk-Out endpoint (as indicated by the
106  // Direction bit) during the execution of this
107  // command.
108  uint8_t bmCBWFlags; // The bits of this field are defined as follows:
109  // Bit 7 : Direction - the device shall ignore
110  // this bit if the dCBWDataTransferLength
111  // field is zero, otherwise:
112  // 0 = Data-Out from host to the device,
113  // 1 = Data-In from the device to the host.
114  // Bit 6 : Obsolete. The host shall set this bit to zero.
115  // Bits 5..0 : Reserved - the host shall set these bits to zero
116  uint8_t bCBWLUN; // The device Logical Unit Number (LUN) to which
117  // the command block is being sent.
118  uint8_t bCBWCBLength; // The valid length of the CBWCB in bytes
119  uint8_t CBWCB[16]; // The command block to be executed by the device.
120 } USB_MSC_CBW;
