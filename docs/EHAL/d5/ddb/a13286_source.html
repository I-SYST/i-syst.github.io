268 typedef struct _USB_Endpoint_Descriptor {
269  uint8_t bLength; // Size of Descriptor in Bytes (7 bytes)
270  uint8_t bDescriptorType; // Endpoint Descriptor (0x05)
271  uint8_t bEndpointAddress; // Endpoint Address
272  // Bits 0..3b Endpoint Number.
273  // Bits 4..6b Reserved. Set to Zero
274  // Bits 7 Direction 0 = Out, 1 = In (Ignored for Control Endpoints)
275  uint8_t bmAttributes; // Bits 0..1 Transfer Type
276  // 00 = Control
277  // 01 = Isochronous
278  // 10 = Bulk
279  // 11 = Interrupt
280  // Bits 2..7 are reserved.
281  // If Isochronous endpoint,
282  // Bits 3..2 = Synchronisation Type (Iso Mode)
283  // 00 = No Synchonisation
284  // 01 = Asynchronous
285  // 10 = Adaptive
286  // 11 = Synchronous
287  // Bits 5..4 = Usage Type (Iso Mode)
288  // 00 = Data Endpoint
289  // 01 = Feedback Endpoint
290  // 10 = Explicit Feedback Data Endpoint
291  // 11 = Reserved
292  uint16_t wMaxPacketSize; // Maximum Packet Size this endpoint is capable of sending or receiving
293  uint8_t bInterval; // Interval for polling endpoint data transfers. Value in frame counts.
294  // Ignored for Bulk & Control Endpoints. Isochronous must equal 1 and
295  // field may range from 1 to 255 for interrupt endpoints.
296 } USB_ENDPOINT_DESC;
