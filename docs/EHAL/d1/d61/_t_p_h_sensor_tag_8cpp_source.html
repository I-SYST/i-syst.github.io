234 void ReadPTHData()
235 {
236  g_pIntrf->Enable();
237 
238  TPHSENSOR_DATA data;
239 
240  g_TphSensor.Read(data);
241  g_TphSensor.StartSampling();
242 
243  g_AdvData.Type = BLEADV_MANDATA_TYPE_TPH;
244 
245  // NOTE : M0 does not access unaligned data
246  // use local 4 bytes align stack variable then mem copy
247  // skip timestamp as advertising pack is limited in size
248  memcpy(&g_TPHData, ((uint8_t*)&data) + 4, sizeof(BLEADV_MANDATA_TPHSENSOR));
249 
250  // Update advertisement data
251  BleAppAdvManDataSet(g_AdvDataBuff, sizeof(g_AdvDataBuff));
252 
253  g_pIntrf->Disable();
254 }
